	//BLOQUE DECLARATIVO
	DEFINIR CARTON, HUEVOS COMO ENTERO
	DEFINIR PRECIOCARTON, PRECIOHUEVO, PRECIOFINAL,AUX COMO REAL
	
	//BLOQUE DE ENTRADAS
	
	
	CARTON = 30 //ASIGNA EL VALOR POR DEFECTO YA QUE GENERALMETE LOS CARTONES TRAE 30 HUEVOS
	MOSTRAR" DEACUERDO AL PRODUCTOR DE HUEVOS SACA CARTONES DE 12, 30 Y 36 HUEVOS"
	MOSTRAR"POR LO GENERAL TRAE 30 HUEVOS EL CARTON"
	
	MOSTRAR "CUANTOS  HUEVOS CONTIENE EL CARTON?"
	LEER CARTON
	
	MOSTRAR "CUANTO CUESTA EL CARTON DE HUEVOS?"
	
	LEER PRECIOCARTON
	
	MOSTRAR "CUANTOS HUEVOS DESEA COMPRAR?"
	
	LEER HUEVOS
	
	
	
	
	//BLOQUE DE PROCESAMIENTO DE LA INFORMACION Y CALCULO
	
	PRECIOHUEVO = PRECIOCARTON / CARTON //cálculo para determinar cuál es el precio de un huevo 
	
	Si HUEVOS < CARTON Entonces
		PRECIOHUEVO = (PRECIOHUEVO * 1.2)
		PRECIOFINAL = PRECIOHUEVO * HUEVOS //el cálculo para determinar cuál es monto que debe pagar
	SiNo
		Si HUEVOS=CARTON Entonces
			PRECIOHUEVO = PRECIOHUEVO
			PRECIOFINAL = PRECIOHUEVO * HUEVOS //el cálculo para determinar cuál es monto que debe pagar
			
		SiNo
			//ESTA PIDIENDO MAS DE UN CARTON, POR LO QUE SE LE VENDE 
			//EL CARTON A SU PRECIO Y LOS SOBRANTES DE CALCULA COMO DETALLADOS
			AUX = HUEVOS - CARTON
			PRECIOHUEVO = (PRECIOHUEVO * 1.2)
			PRECIOFINAL = PRECIOHUEVO * AUX + PRECIOCARTON
		FinSi
	FinSi
	
	
			
	 
	
	
	//BLOQUE DE SALIDA O RESULTADOS FINALES
	
	MOSTRAR "EL PRECIO DEL HUEVO DETALLADO ES ",PRECIOHUEVO," BS" //las salidas del precio del huevo detallado
